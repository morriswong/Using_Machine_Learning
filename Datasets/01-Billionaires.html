# billionaires' total worth by country & industry

total_Billionaires_net_worth_by_country = {"countries": bil["Country"].unique(),"totalNetWorths": []}
total_Billionaires_net_worth_by_industry = {"industries": bil["Industry"].unique(), "totalNetWorths": []}


for country in total_Billionaires_net_worth_by_country["countries"]:
    total_Billionaires_net_worth_by_country["totalNetWorths"].append(bil["Total_net_worth"].loc[bil["Country"] == country].sum())
for industry in total_Billionaires_net_worth_by_industry["industries"]:
    total_Billionaires_net_worth_by_industry["totalNetWorths"].append(bil["Total_net_worth"].loc[bil["Industry"] == industry].sum())

total_Billionaires_net_worth_by_country = pd.DataFrame(total_Billionaires_net_worth_by_country)
total_Billionaires_net_worth_by_industry = pd.DataFrame(total_Billionaires_net_worth_by_industry)
